ALGORITHM IsSafe(currentTenant, targetZone, existingAssignments)

    FOR each assignment IN existingAssignments DO
        otherTenant ← assignment.tenant
        assignedZone ← assignment.zone
        
        // Cek apakah waktu operasional mereka saling tumpang tindih
        timeOverlap ← (currentTenant.startTime < otherTenant.endTime) AND 
                      (otherTenant.startTime < currentTenant.endTime)

        // Evaluasi konflik hanya jika waktunya bentrok
        IF timeOverlap = TRUE THEN
            // 1. Cek Bentrok Fisik
            IF assignedZone = targetZone THEN
                RETURN FALSE // Tidak boleh 2 tenant menempati stand yang sama di jam yang sama
            END IF

            // 2. Cek Bentrok Persaingan (Graph Coloring Constraint)
            IF currentTenant.category = otherTenant.category THEN
                RETURN FALSE // Tidak boleh 2 tenant dengan kategori sama buka di jam yang sama
            END IF

        END IF
    END FOR

    RETURN TRUE
    
END ALGORITHM

// Tujuan: Menjaga integritas jadwal (mencegah double-booking pada satu fisik stand) 
// dan menerapkan batasan Graph Coloring untuk mencegah dua UMKM dengan kategori produk 
// yang sama (misal: "Makanan" dan "Makanan") beroperasi secara bersamaan demi mencegah kanibalisasi pasar.
// Jika waktunya tidak bentrok (beda shift/beda hari), tenant bebas menggunakan stand mana pun.
