ALGORITHM MainProgram

    tenants ← LoadData(SampleData)
    totalZones ← 4

    WHILE TRUE DO
        DISPLAY Menu Utama
        choice ← INPUT user

        SWITCH choice DO
            CASE 1:
                DISPLAY tenants (Tabel Pendaftar)
            CASE 2:
                greedyResult ← GreedyScheduler(tenants, totalZones)
                DISPLAY Ringkasan greedyResult
            CASE 3:
                backtrackResult ← BacktrackingScheduler(tenants, totalZones)
                DISPLAY Ringkasan backtrackResult
            CASE 4:
                START TIMER
                greedyResult ← GreedyScheduler(tenants, totalZones)
                STOP TIMER

                START TIMER
                backtrackResult ← BacktrackingScheduler(tenants, totalZones)
                STOP TIMER

                DISPLAY Komparasi Jadwal & Waktu Eksekusi
            CASE 5:
                extremeTenants ← LoadData(ExtremeData)
                Run Greedy dan Backtracking menggunakan extremeTenants
                DISPLAY Demonstrasi Kelemahan Greedy
            CASE 0:
                EXIT PROGRAM
        END SWITCH
    END WHILE

END ALGORITHM

        Selesai.
