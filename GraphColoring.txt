ALGORITHM BacktrackingScheduler(tenantIndex, zoneAllocation, acceptedCount)

    // OPTIMASI PRUNING: Hentikan pencarian jika sisa tenant tidak akan melampaui rekor terbaik
    remainingTenants ← totalTenants - tenantIndex
    IF acceptedCount + remainingTenants <= bestCount THEN
        RETURN
    END IF

    // BASIS REKURSI: Jika semua tenant sudah dievaluasi
    IF tenantIndex = totalTenants THEN
        IF acceptedCount > bestCount THEN
            bestCount ← acceptedCount
            bestSolution ← buildAssignments(zoneAllocation)
        END IF
        RETURN
    END IF

    // Mencoba tempatkan tenant di setiap zona yang tersedia
    FOR zone ← 1 TO totalZones DO
        IF IsSafe(tenantIndex, zone, zoneAllocation) THEN
            zoneAllocation[tenantIndex] ← zone
            BacktrackingScheduler(tenantIndex + 1, zoneAllocation, acceptedCount + 1)
            zoneAllocation[tenantIndex] ← 0 // Proses Backtracking (Mundur dan reset)
        END IF
    END FOR

    // Opsi alternatif: Melewati (menolak) tenant ini dan lanjut ke tenant berikutnya
    BacktrackingScheduler(tenantIndex + 1, zoneAllocation, acceptedCount)

END ALGORITHM

// Penjelasan & Kompleksitas Waktu:
// Worst Case: O((Z + 1)^N)
// N = jumlah tenant, Z = jumlah zona (stand)

// Setiap tenant memiliki Z pilihan zona ditambah 1 pilihan untuk ditolak (di-skip).
// Algoritma ini dimodifikasi dari Graph Coloring murni untuk mencari "Jumlah Tenant Maksimal" (Global Optimum).

// (zoneAllocation[tenantIndex] ← 0) untuk mengeksplorasi cabang kemungkinan lain.
