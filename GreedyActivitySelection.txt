ALGORITHM GreedyScheduler(tenants, totalZones)

    SORT tenants BY endTime ASCENDING
    assignments ← empty list

    FOR each tenant IN tenants DO
        assigned ← FALSE
        
        FOR zone ← 1 TO totalZones DO
            IF IsSafe(tenant, zone, assignments) THEN
                ADD (tenant, zone) TO assignments
                assigned ← TRUE
                BREAK // Lanjut ke tenant berikutnya setelah dapat zona
            END IF
        END FOR
        
        IF assigned = FALSE THEN
            ADD tenant TO rejectedTenants
        END IF
    END FOR
    
    RETURN assignments

END ALGORITHM

// Penjelasan & Kompleksitas Waktu:
// 1. Urutkan tenant berdasarkan waktu selesai (Earliest Finish Time) agar menyisakan lebih banyak waktu untuk tenant lain.
// 2. Evaluasi tenant satu per satu dan coba masukkan ke zona 1 hingga Z.
// 3. Jika aman (IsSafe = TRUE), langsung tempatkan dan pindah ke tenant berikutnya tanpa mengubah keputusan (Sifat mutlak Greedy).

// Kompleksitas Waktu:
// Sorting   : O(N log N)
// Selection : O(N * Z * M) -> N = tenant, Z = zona, M = jumlah tenant yang sudah diterima
// Total Worst Case didominasi oleh proses Iterasi dan Sorting, menjadikannya sangat cepat secara komputasi dibandingkan Backtracking.
